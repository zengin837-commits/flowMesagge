function initSocket() { socket = io(); socket.on('connect', () => { console.log('Socket baglandi:', socket.id); socket.emit('join', token); }); socket.on('connect_error', (e) => { console.log('Socket hatasi:', e.message); }); socket.on('joined', (d) => { console.log('Odaya katilindi:', d.userId); }); socket.on('qr', (d) => { console.log('QR alindi!'); document.getElementById('qr-img').src = d.qr; document.getElementById('qr-wrap').style.display = 'block'; document.getElementById('qr-hint').textContent = 'WhatsApp → Bagli Cihazlar → Cihaz Ekle → QR tara'; }); socket.on('pairing_code', (d) => { const el = document.getElementById('pairing-code-box'); el.textContent = d.code; el.style.display = 'block'; }); socket.on('connected', () => setWaStatus(true)); socket.on('disconnected', () => setWaStatus(false)); socket.on('send_progress', d => updateProgress(d)); socket.on('send_complete', d => { showToast('Tamamlandi: ' + d.sent + ' basarili, ' + d.failed + ' hatali'); document.getElementById('progress-wrap').style.display = 'none'; document.getElementById('send-btn').disabled = false; document.getElementById('send-btn').textContent = 'Toplu Gonder'; }); }
